2) SELECT k.imie, k.nazwisko, COUNT(g.id_glosu) as Ile_glosow FROM `2017_glosy` g 
JOIN `2017_kandydaci` k ON (k.id_kandydata=g.id_kandydata)
GROUP BY k.id_kandydata 
ORDER BY Ile_glosow  DESC 
LIMIT 10
